print("1.(today im here to do the calculation)")
print("I will take two numbers from you and calculate!")
num1 = input("please insert first number: ")
num2 = input("please insert second number: ")
operation = input("please select your operation between : + , - , * , / , ** , // : ")
if operation == "+":
    result = float(num1) + float(num2)
elif operation == "-":
    result = float(num1) - float(num2)
elif operation == "*":
    result = float(num1) * float(num2)
elif operation == "/":
    result = float(num1) / float(num2)
elif operation == "%":
    result = float(num1) % float(num2)
elif operation == "**":
    result = float(num1) ** float(num2)
elif operation == "//":
    result = float(num1) // float(num2)
else:
    result = "this operation that user inserted is wrong!"
print(result)

exit(0)
